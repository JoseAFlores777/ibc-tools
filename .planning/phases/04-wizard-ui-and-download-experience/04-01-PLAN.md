export const packageRequestSchema = z.object({
  hymns: z.array(z.object({
    id: z.string().uuid(),
    audioFiles: z.array(z.enum(['track_only','midi_file','soprano_voice','alto_voice','tenor_voice','bass_voice'])).optional().default([]),

  layout: z.enum(['one-per-page', 'two-per-page']),
  style: z.enum(['decorated', 'plain']),
});
export type PackageRequest = z.infer<typeof packageRequestSchema>;

    - In `app/api/hymns/search/route.ts`, after parsing `q`, detect if `q` is a numeric string: `const numericQ = q && /^\d+$/.test(q) ? Number(q) : undefined;`
    - Pass both to searchHymns: `{ query: q, hymnNumber: numericQ, hymnalId: hymnal, categoryId: ..., limit, offset }`
    - This means typing "123" searches by name AND by hymn_number simultaneously
    - NOTE: searchHymns uses AND logic for filters. If user types "123" and it matches by number but not by name, the name filter would exclude it. To fix: when `numericQ` is defined, use `_or` filter logic. Simplest approach: if `q` is purely numeric, pass ONLY `hymnNumber` (not query). If `q` contains non-digits, pass ONLY `query`. This matches user intent: typing a number means searching by number.
