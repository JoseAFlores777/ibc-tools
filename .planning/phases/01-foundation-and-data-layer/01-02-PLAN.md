```typescript
import { readItems } from '@directus/sdk';
import { getDirectus } from '@/app/lib/directus';

export async function fetchChurchEvents(options?: { limit?: number }) {
  const client = getDirectus();
  const limit = options?.limit ?? 50;
  const fields = ['id', 'title', ...] as const;
  const items = await client.request(
    // @ts-ignore
    readItems('church_events' as any, {
      sort: ['start_datetime'],
      fields: fields as unknown as string[],
      filter: { status: { _neq: 'archived' } },
      limit,
    })
  );
  return items as ChurchEventListItem[];
}
```

```typescript
async function getHymn(id: string): Promise<ActivityHymn> {
  const queryItem = {
    fields: [
      'name', 'bible_text', 'bible_reference', 'hymn_number', 'hymn_time_signature', 'letter_hymn',
      { hymnal: ['name', 'publisher'],
        authors: ['authors_id.name', 'author_roles.author_roles_id.description', 'author_roles.author_roles_id.rol_abbr'] }
    ],
  };
  const data = await directus.request(readItem<any, any, any>('hymn', id as any, queryItem as any)) as ActivityHymn;
  return data;
}
```
