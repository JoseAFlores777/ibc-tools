```typescript
import { createStreamingZip, assembleHymnPackage } from '@/app/lib/zip/generate-hymn-zip';
import { packageRequestSchema } from '@/app/lib/zip/zip.schema';

export const dynamic = 'force-dynamic';

export async function POST(request: Request) {
  // Step 1: Parse and validate request body with Zod
  let body: unknown;
  try {
    body = await request.json();
  } catch {
    return new Response(
      JSON.stringify({ ok: false, error: 'Request body must be valid JSON' }),
      { status: 400, headers: { 'Content-Type': 'application/json' } },
    );
  }

  const parsed = packageRequestSchema.safeParse(body);
  if (!parsed.success) {
    return new Response(
      JSON.stringify({
        ok: false,
        error: 'Solicitud invalida',
        details: parsed.error.flatten().fieldErrors,
      }),
      { status: 400, headers: { 'Content-Type': 'application/json' } },
    );
  }

  const packageRequest = parsed.data;

  // Step 2: Create streaming ZIP (per D-01, D-02)
  const { archive, webStream } = createStreamingZip();

  // Step 3: Start assembly concurrently -- DO NOT await (per Pitfall 2 from RESEARCH.md)
  // The Response is returned immediately with the stream. Assembly writes entries
  // to the archive which pipes through the already-returned stream.
  const assemblyPromise = assembleHymnPackage(archive, packageRequest)
    .then(({ successCount, errorCount }) => {
      if (successCount === 0) {
        // Per D-09: if ALL items failed, we can't retroactively change the response
        // status (headers already sent). The archive will contain only ERROR.txt files.
        // Log for server-side monitoring.
        console.error(
          `POST /api/hymns/package: all ${errorCount} hymns failed. ZIP contains only error files.`,
        );
      }
    })
    .catch((err) => {
      console.error('POST /api/hymns/package assembly error:', err);
      archive.abort();
    });

  // Step 4: For D-09 edge case -- we need to check if we can detect total failure
  // BEFORE starting to stream. Strategy: do a quick pre-check by counting hymns.
  // If request has 0 hymns (shouldn't pass Zod .min(1) but defensive), return 500.
  // For runtime all-fail: since we stream, we can't change status code after headers.
  // The ERROR.txt files in the ZIP serve as the error indicator.

  // Step 5: Return streaming response immediately (per D-02, D-10, D-11)
  return new Response(webStream, {
    headers: {
      'Content-Type': 'application/zip',
      'Content-Disposition': 'attachment; filename="himnos.zip"',
      'X-Hymn-Count': String(packageRequest.hymns.length),
    },
  });
}
```

IMPORTANT implementation notes:
- Use `new Response()` NOT `NextResponse` for streaming binary (per anti-pattern from RESEARCH.md)
- The `assemblyPromise` fires concurrently -- the Response is returned immediately
- `archive.abort()` is called on fatal assembly errors to close the stream
- Per D-11: `X-Hymn-Count` header set to total hymns count
- Per D-02: `Content-Disposition: attachment; filename="himnos.zip"` triggers browser download
- Per D-10: no SSE or polling; streaming starts immediately, browser shows download progress
- Per D-09: Zod `.min(1)` prevents empty requests. Runtime all-fail case logged server-side; ZIP will contain only ERROR.txt files (status can't change after headers sent for streaming)
